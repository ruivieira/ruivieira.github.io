Hill-climbing optimisation

Updated 2022-07-01 (e7bb158)

Global maximum

Let’s try it with the function

$$ f(x,y) = e^{-\left(x^2+y^2\right)} $$

import numpy as np
import matplotlib.pyplot as plt
from plotutils import *

x = np.linspace(-2.0, 2.0, 1000)
y = np.linspace(-2.0, 2.0, 1000)
X, Y = np.meshgrid(x, y)

Z = np.exp(-(X ** 2 + Y ** 2))

fig, ax = plt.subplots(1, 1)
cp = ax.contourf(X, Y, Z, cmap=cmaps[1])

ax.set_title("f(x,y)")
ax.set_xlabel("x")
ax.set_ylabel("y")
plt.show()


from gradient_free_optimizers import HillClimbingOptimizer

search_space = {
    "x": x,
    "y": y,
}

opt = HillClimbingOptimizer(search_space)

def f(pos):
    x = pos["x"]
    y = pos["y"]
    z = np.exp(-(x**2 + y**2))
    return z

result = opt.search(f,
                    n_iter=30000,
                    verbosity=['print_times'])

Local maximum

Let’s try it with the function

$$ f(x,y) = e^{-\left(x^2+y^2\right)}+2e^{-\left((x-1.7)^2+(y-1.7)^2\right)} $$

x = np.linspace(-1.0, 3.0, 1000)
y = np.linspace(-1.0, 3.0, 1000)
X, Y = np.meshgrid(x, y)

Z = np.exp(-(X**2 + Y**2))+2*np.exp(-((X-1.7)**2+(Y-1.7)**2))

fig,ax=plt.subplots(1,1)
cp = ax.contourf(X, Y, Z, cmap=cmaps[1])

ax.set_title('f(x,y)')
ax.set_xlabel('x')
ax.set_ylabel('y')
plt.show()


opt = HillClimbingOptimizer(search_space)

def f(pos):
    x = pos["x"]
    y = pos["y"]
    z = np.exp(-(x**2 + y**2))+2*np.exp(-((x-1.7)**2+(y-1.7)**2))
    return z

result = opt.search(f,
                    n_iter=30000, 
                    verbosity=['print_times'])
